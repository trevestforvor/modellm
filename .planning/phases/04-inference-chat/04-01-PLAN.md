# Plan 04-01: InferenceService — llama.cpp XCFramework Integration

## Goal

Wire the llama.cpp XCFramework binary into the project and build `InferenceService` as a Swift actor that loads GGUF models and streams tokens via `AsyncThrowingStream`. This is the foundation all other Phase 4 plans depend on.

## Must-Haves

- [ ] llama.cpp XCFramework binary target resolves in Xcode without errors
- [ ] `InferenceService` actor loads a GGUF file from disk and holds one `LlamaSession` resident
- [ ] `generate()` returns `AsyncThrowingStream<String, Error>` and runs inference on `Task.detached` — never on `MainActor`
- [ ] `stopGeneration()` signals the decode loop via actor-isolated boolean — not just `Task.cancel()`
- [ ] `AppContainer` holds `InferenceService` instance
- [ ] Unit tests for `PromptFormatter.chatml()` pass on simulator

## Context

llama.cpp XCFramework binary target is established in `STACK.md`. The API surface must be verified from XCFramework headers before writing the session layer (blocker noted in STATE.md). Use `ChipProfile.contextWindowCap` from `CompatibilityModels.swift` to configure `n_ctx`.

```swift
import Foundation
import OSLog

private let logger = Logger(subsystem: "com.modelrunner", category: "InferenceService")

/// Manages llama.cpp model loading and token-streaming inference.
/// Actor isolation prevents concurrent access to llama_context.
/// All generation runs on Task.detached — never on MainActor.
actor InferenceService {
    private var session: LlamaSession?
    private var isCancelled: Bool = false

    // MARK: - Model Management

    /// Load a GGUF model from disk. Expensive operation (2-30 seconds).
    /// Call from ChatViewModel before the first generate() call.
    func loadModel(at url: URL, params: InferenceParams) async throws {
        // Unload existing model first
        if session != nil {
            await unloadModel()
        }
        isCancelled = false
        logger.info("Loading model: \(url.lastPathComponent)")
        session = try LlamaSession(modelURL: url, params: params)
        logger.info("Model loaded: \(url.lastPathComponent)")
    }

    func unloadModel() async {
        session = nil  // triggers LlamaSession.deinit which calls llama_free
        logger.info("Model unloaded")
    }

    var isLoaded: Bool { session != nil }

    // MARK: - Inference

    /// Returns a stream of token strings. Runs inference on Task.detached.
    /// The caller (ChatViewModel) iterates on a non-MainActor task.
    func generate(prompt: String) -> AsyncThrowingStream<String, Error> {
        AsyncThrowingStream { [weak self] continuation in
            guard let self else {
                continuation.finish(throwing: InferenceError.noActiveSession)
                return
            }
            Task.detached(priority: .userInitiated) {
                guard let session = await self.session else {
                    continuation.finish(throwing: InferenceError.noActiveSession)
                    return
                }
                await self.setNotCancelled()
                do {
                    // Tokenize prompt
                    // Run decode loop: for each token:
                    //   if await self.isCancelled { continuation.finish(); return }
                    //   let piece = llama_token_to_piece(token)
                    //   if piece.isEmpty || token == eosToken { continuation.finish(); return }
                    //   continuation.yield(piece)
                    continuation.finish()
                } catch {
                    continuation.finish(throwing: error)
                }
            }
        }
    }

    /// Signal the decode loop to stop after the current token.
    func stopGeneration() {
        isCancelled = true
    }

    // MARK: - Private

    private func setNotCancelled() {
        isCancelled = false
    }
}
```

```swift
import Foundation
import OSLog

@Observable
final class AppContainer {
    let deviceService = DeviceCapabilityService()
    let inferenceService = InferenceService()
    private(set) var compatibilityEngine: CompatibilityEngine?

    init() {
        Task {
            await deviceService.initialize()
            if let specs = await deviceService.specs {
                self.compatibilityEngine = CompatibilityEngine(device: specs)
            }
        }
    }

    /// Convenience: build InferenceParams from the device's ChipProfile.
    /// Falls back to conservative defaults if device profile not yet loaded.
    func inferenceParams() -> InferenceParams {
        let contextCap = compatibilityEngine?.deviceSpecs.contextWindowCap ?? 2048
        return InferenceParams.default(contextWindowCap: contextCap)
    }
}
```

```swift
import XCTest
@testable import ModelRunner

final class PromptFormatterTests: XCTestCase {

    func testChatmlEmptyMessages() {
        let result = PromptFormatter.chatml(system: "You are helpful.", messages: [])
        XCTAssertTrue(result.hasPrefix("<|im_start|>system\nYou are helpful.<|im_end|>\n"))
        XCTAssertTrue(result.hasSuffix("<|im_start|>assistant\n"))
    }

    func testChatmlUserMessage() {
        let messages = [ChatMessage(role: .user, content: "Hello")]
        let result = PromptFormatter.chatml(system: "Be helpful.", messages: messages)
        XCTAssertTrue(result.contains("<|im_start|>user\nHello<|im_end|>\n"))
        XCTAssertTrue(result.hasSuffix("<|im_start|>assistant\n"))
    }

    func testChatmlConversation() {
        let messages = [
            ChatMessage(role: .user, content: "What is 2+2?"),
            ChatMessage(role: .assistant, content: "4"),
            ChatMessage(role: .user, content: "Why?")
        ]
        let result = PromptFormatter.chatml(system: "Math tutor.", messages: messages)
        XCTAssertTrue(result.contains("<|im_start|>assistant\n4<|im_end|>\n"))
        XCTAssertTrue(result.hasSuffix("<|im_start|>assistant\n"))
    }

    func testChatmlRoleOrder() {
        // Ensure system always comes first
        let messages = [ChatMessage(role: .user, content: "Hi")]
        let result = PromptFormatter.chatml(system: "sys", messages: messages)
        let systemRange = result.range(of: "<|im_start|>system")!
        let userRange = result.range(of: "<|im_start|>user")!
        XCTAssertLessThan(systemRange.lowerBound, userRange.lowerBound)
    }
}
```

```swift
        XCTAssertFalse(loaded)
    }

    func testLoadNonexistentModelThrows() async {
        let service = InferenceService()

        let params = InferenceParams.default(contextWindowCap: 2048)
        do {
            try await service.loadModel(at: fakeURL, params: params)
            XCTFail("Expected modelLoadFailed error")
        } catch InferenceError.modelLoadFailed {
            // expected
        } catch {
            XCTFail("Unexpected error: \(error)")
        }
    }

    func testGenerateWithoutModelThrows() async {
        let service = InferenceService()
        var threwError = false
        do {
            for try await _ in await service.generate(prompt: "hello") {
                XCTFail("Should not yield tokens without loaded model")
            }
        } catch InferenceError.noActiveSession {
            threwError = true
        } catch {
            XCTFail("Unexpected error: \(error)")
        }
        XCTAssertTrue(threwError)
    }

    func testUnloadAfterLoad() async throws {
        // This test verifies state transitions only — no real model file needed
        let service = InferenceService()
        await service.unloadModel()  // Should not crash
        let loaded = await service.isLoaded
        XCTAssertFalse(loaded)
    }
}
```
